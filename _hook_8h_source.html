381 template<typename ReturnType, typename... Args>
382 ReturnType callPlugins(HookedCall target, HookStep step, bool& skipFunctionCall, Args&&... args) const
383 {
384 using PluginCallType = ReturnType(Args...);
385 constexpr bool ReturnTypeIsVoid = std::is_same_v<ReturnType, void>;
386 using NoVoidReturnType = std::conditional_t<ReturnTypeIsVoid, int, ReturnType>;
387
388 NoVoidReturnType ret {};
389 TRY_HOOK
390 {
391 for (const auto& hook : pluginHooks_[uint(target) * uint(HookStep::Count) + uint(step)])
392 {
393 const auto& plugin = hook.plugin();
394 if (plugin.paused)
395 continue;
396
397 if (plugin.resetCode)
398 *plugin.returnCode = ReturnCode::Default;
399
400 TRY_HOOK
401 {
402 if constexpr (ReturnTypeIsVoid)
403 reinterpret_cast<PluginCallType*>(hook.hookFunction)(std::forward<Args>(args)...);
404 else
405 ret = reinterpret_cast<PluginCallType*>(hook.hookFunction)(std::forward<Args>(args)...);
406 }
407 CATCH_HOOK({
408 AddLog(
409 Normal, L"ERROR: Exception in plugin '%s' in %s", stows(plugin.name).c_str(),
410 stows(__FUNCTION__).c_str());
411 });
412
413 auto code = *plugin.returnCode;
414
415 if ((code & ReturnCode::SkipFunctionCall) != ReturnCode::Default)
416 skipFunctionCall = true;
417
418 if ((code & ReturnCode::SkipPlugins) != ReturnCode::Default)
419 break;
420 }
421 }
422 CATCH_HOOK({ AddLog(Normal, L"ERROR: Exception %s", stows(__FUNCTION__).c_str()); });
423
424 if constexpr (!ReturnTypeIsVoid)
425 return ret;
426 }
427};

429template<typename ReturnType = void, typename... Args>
430auto CallPluginsBefore(HookedCall target, Args&&... args)
431{
432 bool skip = false;
433 if constexpr (std::is_same_v<ReturnType, void>)
434 {
435 PluginManager::i()->callPlugins<void>(target, HookStep::Before, skip, std::forward<Args>(args)...);
436 return skip;
437 }
438 else
439 {
440 ReturnType ret =
441 PluginManager::i()->callPlugins<ReturnType>(target, HookStep::Before, skip, std::forward<Args>(args)...);
442 return std::make_tuple(ret, skip);
443 }
444}
